void swap(int *px,int *py); /*a function prototype*/

int main(){
    int a=5,b=10;
    printf("before swapping a=%d,b=%d",a,b);
    swap(&a,&b);
    printf("after swapping a=%d,b=%d",a,b);
    return 0;
}
void swap(int *px,int *py)/*a function to swap the values*/
{
    int temp;
    temp=*px;
    *px=*py;
    *py=temp;
}#include<stdio.h>

/*this is a function pointer*/
void(*operation)(int,int);

void add(int a,int b){
    printf("sum:%d\n",a+b);
}
void subtract(int a,int b){
    printf("Difference:%d",a-b);
}
int main(){
    operation=add;
    int num1,num2;
    printf("Enter num1 and num2:\n");
    scanf("%d %d",&num1,&num2);
    operation(num1,num2);

    operation=subtract;
    printf("Enter num1 and num2:\n");
    scanf("%d %d",&num1,&num2);
    operation(num1,num2);

    return 0;
}
