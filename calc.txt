#include<stdio.h>
#include<stdlib.h>
#include<math.h>
int main(){
    char operator;
    double num1,num2,result;

    printf("WELCOME TO CALCULATOR:\t\n");
    printf("Now enter the operator(+,-,*,/):");
    scanf(" %c",&operator);


    printf("Enter the two numbers:");
    scanf("%lf %lf",&num1,&num2);

    switch(operator){
        case '+':
        result =num1 + num2;
        printf("%lf\n",result);
        break;
        case '-':
        result =num1-num2;
        printf("%lf\n",result);
        break;
        case '*':
        result =num1*num2;
        printf("%lf\n",result);
        break;
        case '/':
        if(num2!=0){
            result =num1/num2;
        printf("%lf\n",result);
        }
        else{
            printf("Error cannot divide by zero\n");
        }
        break;
        default:
        printf("Invalid operator!");
        break;
    }   
    return 0;
}
